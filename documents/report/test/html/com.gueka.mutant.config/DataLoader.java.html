package com.gueka.mutant.config;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.boot.CommandLineRunner;
import org.springframework.stereotype.Component;

import com.gueka.mutant.model.DNA;
import com.gueka.mutant.repository.DNARepository;

@Component
public class DataLoader implements CommandLineRunner {

	private static final Logger log = LoggerFactory.getLogger(DataLoader.class);

	
	@Autowired
	private DNARepository repository;
	
    @Override
    public void run(final String... args) throws Exception {
        if (repository.count().block() == 0L) {
    		DNA dna = new DNA();
    		dna.setDna(new String[]{"ATGCGA","CAGTGC","TTATGT","AGAAGG","CCCCTA","TCACTG"});
    		dna.setIsMutant(true);
    		repository.save( dna ).block();
        }
    }
    
}
